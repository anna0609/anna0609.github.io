```{r codes, include=FALSE}
data("instacart") 
instacart %>% 
  count(aisle) %>% 
  arrange(desc(n)) %>% 
  filter(n>10000) %>% 
  mutate(
    aisle = factor(aisle),
    aisle = fct_reorder(aisle,n)
    )

instacart %>%
  filter(aisle == "milk") %>% 
  group_by(product_name) %>% 
  count(product_name) %>% 
  arrange(desc(n))
```

### Chart C

```{r}
instacart %>% 
  count(aisle) %>% 
  filter(n>10000) %>% 
  mutate(
    aisle = factor(aisle),
    aisle = fct_reorder(aisle,n)
  ) %>% 
  plot_ly(x = ~aisle, y = ~n, color = ~aisle,
          type = "bar", colors = "viridis") %>% 
    layout(
      xaxis = list(title = FALSE),
    title = "Aisles more than 10000 ordered"
  )
```
